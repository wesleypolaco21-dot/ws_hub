local _arConns={}
local function cleanRag(char)
    if not char then return end
    for _,d in ipairs(char:GetDescendants()) do
        if d:IsA("BallSocketConstraint") or d:IsA("NoCollisionConstraint") or d:IsA("HingeConstraint") then
            pcall(function() d:Destroy() end)
        elseif d:IsA("Motor6D") then d.Enabled=true
        elseif d:IsA("BodyVelocity") or d:IsA("BodyPosition") or d:IsA("BodyGyro") then
            pcall(function() d:Destroy() end) end end
    local hum=char:FindFirstChildOfClass("Humanoid")
    if hum then pcall(function() workspace.CurrentCamera.CameraSubject=hum end) end end
local function hookAntiRag(char)
    for _,c in ipairs(_arConns) do pcall(function() c:Disconnect() end) end; _arConns={}
    local hum=char:WaitForChild("Humanoid",8); local hrp=char:WaitForChild("HumanoidRootPart",8)
    if not hum or not hrp then return end
    local c1=hum.StateChanged:Connect(function()
        local st=hum:GetState()
        if st==Enum.HumanoidStateType.Physics or st==Enum.HumanoidStateType.Ragdoll or st==Enum.HumanoidStateType.FallingDown then
            hum:ChangeState(Enum.HumanoidStateType.Running); cleanRag(char) end end)
    local c2=char.DescendantAdded:Connect(function(d)
        if d:IsA("BallSocketConstraint") or d:IsA("NoCollisionConstraint") then
            task.defer(function() cleanRag(char) end) end end)
    local c3=RunService.Heartbeat:Connect(function()
        local st=hum:GetState()
        if st==Enum.HumanoidStateType.Physics or st==Enum.HumanoidStateType.Ragdoll or st==Enum.HumanoidStateType.FallingDown then
            cleanRag(char) end end)
    table.insert(_arConns,c1); table.insert(_arConns,c2); table.insert(_arConns,c3); cleanRag(char) end

    if LocalPlayer.Character then task.spawn(function() hookAntiRag(LocalPlayer.Character) end) end
    LocalPlayer.CharacterAdded:Connect(function(c) task.spawn(function() hookAntiRag(c) end) end) end

local _dsActive,_dsHook,_rn=false,false,nil
local function _findRN()
    if typeof(ranknet)=="table" then return ranknet end
    if typeof(raknet)=="table" then return raknet end
    local ok,fe=pcall(getfenv)
    if ok and fe then
        if typeof(fe.ranknet)=="table" then return fe.ranknet end
        if typeof(fe.raknet)=="table" then return fe.raknet end end
    if typeof(syn)=="table" and typeof(syn.raknet)=="table" then return syn.raknet end; return nil end
local function _rnSend(pkt)
    if not _dsActive then return end; local id=pkt.PacketId or pkt.Id or pkt.ID
    if id~=0x1B then return end
    local ok=pcall(function() local b=pkt.AsBuffer
        buffer.writeu32(b,1,0xFFFFFFFF); buffer.writeu32(b,5,0xFFFFFFFF); buffer.writeu32(b,9,0xFFFFFFFF); pkt:SetData(b) end)
    if not ok then pcall(function()
        local r=pkt.Data or pkt.Buffer or ""; if #r<13 then return end
        local p=r:sub(1,1).."ÿÿÿÿ".."ÿÿÿÿ".."ÿÿÿÿ"..r:sub(14)
        if pkt.SetData then pkt:SetData(p) else pkt.Data=p end end) end end
local function _initHook()
    if _dsHook then return end; _rn=_findRN(); if not _rn then return end
    local function dropRecv(pkt) if not _dsActive then return end
        local id=pkt.PacketId or pkt.Id or pkt.ID
        if id==0x1B or id==0x86 then pcall(function() pkt:Drop() end) end end
    if typeof(_rn.desync)=="function" then _dsHook=true
        pcall(function() _rn.desync(true) end)
        if typeof(_rn.hook)=="function" then pcall(function() _rn.hook(dropRecv) end) end; return end
    if typeof(_rn.add_send_hook)=="function" then _dsHook=true
        pcall(function() _rn.add_send_hook(_rnSend) end)
        if typeof(_rn.add_recv_hook)=="function" then pcall(function() _rn.add_recv_hook(dropRecv) end) end; return end
    if typeof(_rn.on_send)=="function" then _dsHook=true
        pcall(function() _rn.on_send(_rnSend) end)
        if typeof(_rn.on_recv)=="function" then pcall(function() _rn.on_recv(dropRecv) end) end; return end
    for _,k in ipairs({"hook","hook_packet","hookpacket","add_hook"}) do
        local fn=rawget(_rn,k); if typeof(fn)=="function" then _dsHook=true; pcall(function() fn(_rnSend) end); return end end end
local function setDesync(on)
    _dsActive=on; Config.DesyncEnabled=on; SaveConfig()
    if on and not _dsHook then pcall(_initHook) end
    Notif("DESYNC",on and (_dsHook and "ATIVADO" or "SEM RAKNET/RANKNET") or "DESATIVADO") end
pcall(_initHook); if Config.DesyncEnabled then _dsActive=true end

            if pos then local d=(pos-sp).Magnitude; if d<bestD then bestD=d; best=desc end end end end
    if best then PromptCache[a.uid]=best end; return best end

local function buildCB(prompt)
    if InternalCache[prompt] then return end
    local data={hold={},trigger={},ready=true}
    local ok1,c1=pcall(getconnections,prompt.PromptButtonHoldBegan)
    if ok1 and type(c1)=="table" then for _,c in ipairs(c1) do if type(c.Function)=="function" then table.insert(data.hold,c.Function) end end end
    local ok2,c2=pcall(getconnections,prompt.Triggered)
    if ok2 and type(c2)=="table" then for _,c in ipairs(c2) do if type(c.Function)=="function" then table.insert(data.trigger,c.Function) end end end
    if #data.hold>0 or #data.trigger>0 then InternalCache[prompt]=data end end

local function doSteal(prompt)
    if not prompt or not prompt.Parent then return false end
    buildCB(prompt); local data=InternalCache[prompt]
    if not data or not data.ready then return false end; data.ready=false
    task.spawn(function()
        for _,fn in ipairs(data.hold) do task.spawn(fn) end; task.wait(0.8)
        for _,fn in ipairs(data.trigger) do task.spawn(fn) end; data.ready=true end)
    return true end

local STEAL_R=15; local PRED_T=0.6
local function canSteal(a)
    local part=findAdornee(a); if not part then return false end
    local hrp=LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart"); if not hrp then return false end
    local dist=(hrp.Position-part.Position).Magnitude; if dist<=STEAL_R then return true end
    local toT=(part.Position-hrp.Position); local dot=_plrVel:Dot(toT.Unit)
    if dot>0 and toT.Magnitude/math.max(dot,0.1)<=PRED_T then return true end; return false end

local function getBest()
    if #allAnimals==0 then return nil end; local mode=Config.StealMode
    if mode=="priority" then
        for _,pn in ipairs(PRIORITY_LIST) do local pl=pn:lower()
            for _,a in ipairs(allAnimals) do if a.name:lower()==pl then return a end end end
        return allAnimals[1] end
    if mode=="nearest" then
        local hrp=LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        if not hrp then return allAnimals[1] end
        local best,bestD=nil,math.huge
        for _,a in ipairs(allAnimals) do local p=findAdornee(a)
            if p then local d=(hrp.Position-p.Position).Magnitude; if d<bestD then bestD=d; best=a end end end
        return best or allAnimals[1] end
    return allAnimals[1] end
